Project 2: The Folder You've Avoided

Methodology: 4-Step Safety Ritual (Copy, Dry Run, Read, Approve)

1. Safety Workflow & Data Analysis Report

High-volume unorganized folders carry severe data loss risks during automated script execution. This architecture implements a strict execution guard using the 4-Step Safety Ritual:

  • The Copy: Ensuring a localized or duplicate environment exists before running any code.
  • The Dry Run: The script sets a strict dry_run=True flag that reads and matches the files but blocks any actual disk modifications.
  • The Read: Displays a real-time console matrix mapping exactly what actions would happen under full authorization.
  • Catch and Approve: Demands human confirmation by setting the flag to False before making real, permanent structural changes.

2. Desktop Automation Engine (Safe Production Script)


import os
import shutil

def process_avoided_folder(target_dir, dry_run=True):
    """
    Applies the full safety ritual workflow to messy folder volumes.
    """
    print(f"=== SAFETY METRIC INITIATED (Dry Run Mode: {dry_run}) ===")
    if not os.path.exists(target_dir):
        print(f"Error: Target path '{target_dir}' does not exist.")
        return False

    all_assets = os.listdir(target_dir)
    simulated_moves = 0
    actual_moves = 0

    for asset in all_assets:
        asset_path = os.path.join(target_dir, asset)
        if os.path.isdir(asset_path) or asset == "run_safety_workflow.py":
            continue
            
        name, ext = os.path.splitext(asset)
        clean_ext = ext.lower().strip(".")
        if not clean_ext:
            continue

        target_folder_name = f"{clean_ext.upper()} Files"
        destination_path = os.path.join(target_dir, target_folder_name)

        if dry_run:
            print(f"[SIMULATION] WOULD MOVE: {asset} -> {target_folder_name}/")
            simulated_moves += 1
        else:
            if not os.path.exists(destination_path):
                os.makedirs(destination_path)
            shutil.move(asset_path, os.path.join(destination_path, asset))
            print(f"[LIVE MUTATION] Executed asset transfer: {asset} -> {target_folder_name}/")
            actual_moves += 1

    print("\n=== WORKFLOW METRIC SUMMARY ===")
    if dry_run:
        print(f"[READ CONFIRMATION]: Dry run successfully simulated {simulated_moves} asset movements. Zero data was altered.")
    else:
        print(f"[COMPLETED]: Safely migrated {actual_moves} structural records based on human approval.")
    return True

if __name__ == '__main__':
    # Local environment run
    active_directory = os.path.dirname(os.path.abspath(__file__))
    process_avoided_folder(active_directory, dry_run=True)
