Project 3: The Script That Rewrites Itself

Methodology: Dynamic Self-Configuration Engine (JSON Parameter Mapping)

1. Configuration & Automation Report

Hardcoding settings directly into production environments causes systemic brittle failure modes. This architecture detaches runtime logic from hardcoded constraints by reading external states dynamically:

  • Dynamic State Adjustment: The program tracks path metrics and folder names externally without manually rewriting the source script block.
  • State Storage (config.json): A decoupled configuration file stores folder path keys and operational preferences securely.
  • Automation Scalability: Shifts structural targets instantly across local desktops, making the pipeline highly maintainable for future data expansion.

3. Production Engine (Self-Configuring Python Script)


import os
import json
import shutil

def load_system_config():
    """
    Generates or reads external state parameters dynamically.
    """
    config_file = "config.json"
    default_config = {
        "target_directory": ".",
        "safe_mode": True
    }
    
    if not os.path.exists(config_file):
        with open(config_file, 'w') as f:
            json.dump(default_config, f, indent=4)
        print("[CONFIG] Native config.json missing. Generated default template successfully.")
        return default_config
        
    with open(config_file, 'r') as f:
        return json.load(f)

def run_self_configured_pipeline():
    """
    Executes structural operations based on runtime configuration metrics.
    """
    config = load_system_config()
    target_dir = config.get("target_directory", ".")
    safe_mode = config.get("safe_mode", True)
    
    print(f"=== PIPELINE RUNNING FROM CONFIG (Safe Mode: {safe_mode}) ===")
    
    if not os.path.exists(target_dir):
        print(f"Error: Specified directory '{target_dir}' does not exist.")
        return False
        
    for item in os.listdir(target_dir):
        item_path = os.path.join(target_dir, item)
        if os.path.isdir(item_path) or item in ["config.json", "run_pipeline.py"]:
            continue
            
        name, ext = os.path.splitext(item)
        clean_ext = ext.lower().strip(".")
        if not clean_ext:
            continue
            
        folder_title = f"{clean_ext.upper()} Automated Files"
        if safe_mode:
            print(f"[CONFIG CONTROL] Simulating path migration: {item} -> {folder_title}/")
        else:
            dest_dir = os.path.join(target_dir, folder_title)
            if not os.path.exists(dest_dir):
                os.makedirs(dest_dir)
            shutil.move(item_path, os.path.join(dest_dir, item))
            print(f"[CONFIG DEPLOYED] Executed asset transfer: {item} -> {folder_title}/")
            
    return True

if __name__ == '__main__':
    run_self_configured_pipeline()
