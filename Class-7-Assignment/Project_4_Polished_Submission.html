import os
import shutil
import logging

def initialize_telemetry_logger():
    """
    Sets up a persistent, enterprise-grade file logging ecosystem.
    """
    logging.basicConfig(
        filename='automation.log',
        filemode='a',
        format='%(asctime)s - %(levelname)s - %(message)s',
        level=logging.INFO
    )
    # Return a stream logger pointer for terminal echo visibility
    console = logging.StreamHandler()
    console.setLevel(logging.INFO)
    formatter = logging.Formatter('%(asctime)s - %(levelname)s - %(message)s')
    console.setFormatter(formatter)
    logging.getLogger('').addHandler(console)

def run_production_polished_pipeline(target_dir):
    """
    Executes resilient data file organization with active runtime error capture.
    """
    initialize_telemetry_logger()
    logging.info("=== PRODUCTION RESILIENT PIPELINE RUNNING ===")
    
    if not os.path.exists(target_dir):
        logging.error(f"Execution terminated. Provided path does not exist: '{target_dir}'")
        return False
        
    try:
        all_items = os.listdir(target_dir)
        logging.info(f"Directory index scanned successfully. Total assets located: {len(all_items)}")
    except Exception as e:
        logging.error(f"Critical System Fault: Unable to index file structures. Reason: {e}")
        return False

    processed_records = 0
    
    for item in all_items:
        item_path = os.path.join(target_dir, item)
        if os.path.isdir(item_path) or item in ["automation.log", "run_polished_pipeline.py"]:
            continue
            
        try:
            name, ext = os.path.splitext(item)
            clean_ext = ext.lower().strip(".")
            if not clean_ext:
                continue
                
            folder_title = f"{clean_ext.upper()} Polished Files"
            destination_dir = os.path.join(target_dir, folder_title)
            
            if not os.path.exists(destination_dir):
                os.makedirs(destination_dir)
                logging.info(f"Allocated new structural directory branch: {folder_title}/")
                
            shutil.move(item_path, os.path.join(destination_dir, item))
            logging.info(f"[SUCCESS] Migrated record: {item} -> {folder_title}/")
            processed_records += 1
            
        except PermissionError:
            logging.error(f"[ACCESS FAULT] Permission Denied for file asset: '{item}'. Skipping step safely.")
        except Exception as error:
            logging.error(f"[UNKNOWN ERROR] Handled fault on item '{item}'. Reason: {error}")

    logging.info(f"=== PIPELINE COMPLETE: Organized {processed_records} records successfully ===")
    return True

if __name__ == '__main__':
    active_workspace = os.path.dirname(os.path.abspath(__file__))
    run_production_polished_pipeline(active_workspace)
